data <- data %>% select(ctrcode,country, distributive, sex, age, educ, income, selfemploy, unnemployment, platform, tecnoestres, pos_tecnoestres, skills, deservingness, gdp_2024, welfare_regime)

data <- na.omit(data)
reg1 <- lmer(distributive ~ sex + age + educ + income + selfemploy + platform + tecnoestres + pos_tecnoestres + skills + (1|ctrcode), data=data)
reg2 <- lmer(distributive ~ sex + age + educ + income + selfemploy + platform + tecnoestres + pos_tecnoestres + skills + unnemployment + (1|ctrcode), data=data)

reg5 <-lmer(distributive ~ sex + age + educ + income + selfemploy + platform + tecnoestres + pos_tecnoestres + skills + unnemployment + deservingness + gdp_2024 + welfare_regime+ (1|ctrcode), data=data)

reg7 <- lmer(distributive ~ sex + age + educ + income + selfemploy + platform + tecnoestres + pos_tecnoestres*platform + skills + unnemployment + deservingness + gdp_2024 + welfare_regime + (1|ctrcode), data=data)
reg8 <- lmer(distributive ~ sex + age + educ + income + selfemploy + platform + tecnoestres + pos_tecnoestres + skills*platform + unnemployment + deservingness + gdp_2024 + welfare_regime + (1|ctrcode), data=data)
reg9 <- lmer(distributive ~ sex + age + educ + income + selfemploy + platform*selfemploy + tecnoestres + pos_tecnoestres + skills + unnemployment + deservingness + gdp_2024 + welfare_regime + (1|ctrcode), data=data)
