# ============================================================
# Clasificación de regímenes de bienestar — versión revisada
# Fuentes principales:
#   Esping-Andersen (1990) Three Worlds of Welfare Capitalism
#   Ferrera (1996) Southern European regime
#   Aidukaite (2011) Post-communist welfare reforms
#   Fenger (2007) CEE typology
# ============================================================

data <- data %>%
  mutate(
    welfare_regime = case_when(

      # 1. SOCIALDEMÓCRATA
      # Esping-Andersen (1990): universalismo, alta desmercantilización
      country %in% c(5,   # Denmark
                     7,   # Finland
                     19,  # Norway
                     18   # Netherlands
                     ) ~ "Social-democratic",

      # 2. LIBERAL
      # Esping-Andersen (1990): residual, focalizado, mercado
      # Martínez Franzoni (2008): Chile y México — liberal latinoamericano
      # Aidukaite (2011): bálticos con orientación liberal post-soviética
      country %in% c(3,   # Canada
                     11,  # Ireland
                     26,  # United Kingdom
                     27,  # United States
                     4,   # Chile
                     17,  # Mexico
                     6,   # Estonia
                     15,  # Latvia
                     16   # Lithuania
                     ) ~ "Liberal",

      # 3. CONSERVADOR-CORPORATIVISTA + SUR DE EUROPA
      # Esping-Andersen (1990) + Ferrera (1996)
      country %in% c(1,   # Austria
                     2,   # Belgium
                     8,   # France
                     9,   # Germany
                     24,  # Switzerland
                     10,  # Greece
                     13,  # Italy
                     21,  # Portugal
                     23,  # Spain
                     12,  # Israel
                     14,  # Korea
                     25,   # Türkiye
                     20,  # Poland
                     22   # Slovenia
                     ) ~ "Conservative",

      TRUE ~ NA_character_
    ),
    welfare_regime = factor(welfare_regime,
                            levels = c("Liberal",
                                       "Social-democratic",
                                       "Conservative"))
  )
